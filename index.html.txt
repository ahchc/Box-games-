<script>
// إعداد الفائزين
const winnerBoxes = [4, 12, 25, 37]; // 4 صناديق فائزة
let clickedBoxes = [];

for (let i = 1; i <= 40; i++) {
  let box = document.createElement("div");
  box.className = "box";
  box.innerText = "صندوق " + i;

  box.onclick = function() {
    if (clickedBoxes.includes(i)) return;
    clickedBoxes.push(i);

    if (winnerBoxes.includes(i)) {
      alert("🎉 مبرووك لقد فزت!");
    } else {
      alert("❌ ما حالفك الحظ 😅");
    }

    // قفل جميع الصناديق بعد الضغط
    document.querySelectorAll(".box").forEach(b => b.classList.add("disabled"));
  };

  document.getElementById("grid").appendChild(box);
}
</script>
